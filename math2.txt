using namespace std;


enum enQuestionLevel { Easy = 1, Medium, Hard, Mix };

enum enOperationType { Add = 1, Subtract, Multi, Divide, MixOp };

struct stQuestion
{
	int Number1 = 0;
	int Number2 = 0;
	enQuestionLevel QuestionLevel;
	enOperationType Operation;
	int CorrectAnswer = 0;
	int USerGuess = 0;
	bool QuestionResult = false;
};

struct stQuizz
{
	stQuestion QuestionList[100];
	int NumberOfQuestions = 0;
	int NumberOfCorrect = 0;
	int NumberOfWrong = 0;
	enOperationType Operation;
	enQuestionLevel QuestionLevel;
	bool FinalResult = false;
};

short ReadHowManyQuestions()
{
	short NumberOfQuestions = 0;

	do
	{
		cout << "How Many Questions 1 to 10? ";
		cin >> NumberOfQuestions;
	} while (NumberOfQuestions < 1 || NumberOfQuestions > 10);

	return NumberOfQuestions;
}

enQuestionLevel QuestionLevel()
{
	short QuestionLevel = 0;
	do
	{
		cout << "Chose QLevel -> [1]Easy  [2]Medium  [3]Hard  [4]Mix ? ";
		cin >> QuestionLevel;
	} while (QuestionLevel < 1 || QuestionLevel > 4);

	return (enQuestionLevel)QuestionLevel;
}

enOperationType OperationType()
{
	short OpType = 0;

	do
	{
		cout << "Chose OpType -> [1]Add  [2]Subtract  [3]Multi  [4]Divide  [5]MixOp ? ";
		cin >> OpType;

	} while (OpType < 1 || OpType > 5);

	return (enOperationType)OpType;
}

int ReadUserGuess()
{
	int UserGuess = 0;
	cin >> UserGuess;

	return UserGuess;
}

string GetQuestionLevelText(enQuestionLevel Level)
{
	string arrQuestionLevel[4] = { "Easy", "Medium", "Hard", "Mix" };
	return arrQuestionLevel[Level - 1];
}

string GetOpTypeSymbol(enOperationType OpType)
{
	switch (OpType)
	{
	case enOperationType::Add:
		return "+";
	case enOperationType::Subtract:
		return "-";
	case enOperationType::Multi:
		return "*";
	case enOperationType::Divide:
		return "/";
	default:
		return "Unknown";
	}
}

enOperationType GetRandomOperation()
{
	return (enOperationType)MyLib::RandomNumber(1, 4);
}

string GetOpTypeName(enOperationType OpType)
{
	string arrOpType[5] = { "Add", "Subtract", "Multiplication", "Divide", "MixOp" };
	return arrOpType[OpType - 1];
}

int SimpleCalcuLater(enOperationType OpType, int Number1, int Number2)
{
	switch (OpType)
	{
	case enOperationType::Add:
		return Number1 + Number2;
	case enOperationType::Subtract:
		return Number1 - Number2;
	case enOperationType::Multi:
		return Number1 * Number2;
	case enOperationType::Divide:
		if (Number2 == 0) return 0;
		return Number1 / Number2;
	default:
		return Number1 + Number2;
	}
}

stQuestion GenerateQuestion(enOperationType OpType, enQuestionLevel QuestionLevel)
{
	stQuestion Question;

	if (QuestionLevel == enQuestionLevel::Mix)
		QuestionLevel = (enQuestionLevel)MyLib::RandomNumber(1, 3);

	if (OpType == enOperationType::MixOp)
		OpType = GetRandomOperation();

	Question.Operation = OpType;

	switch (QuestionLevel)
	{
	case enQuestionLevel::Easy:
		Question.Number1 = MyLib::RandomNumber(1, 10);
		Question.Number2 = MyLib::RandomNumber(1, 10);

		Question.CorrectAnswer = SimpleCalcuLater(Question.Operation, Question.Number1, Question.Number2);
		Question.QuestionLevel = QuestionLevel;
		break;

	case enQuestionLevel::Medium:
		Question.Number1 = MyLib::RandomNumber(20, 50);
		Question.Number2 = MyLib::RandomNumber(20, 50);

		Question.CorrectAnswer = SimpleCalcuLater(Question.Operation, Question.Number1, Question.Number2);
		Question.QuestionLevel = QuestionLevel;
		break;

	case enQuestionLevel::Hard:
		Question.Number1 = MyLib::RandomNumber(50, 100);
		Question.Number2 = MyLib::RandomNumber(50, 100);

		Question.CorrectAnswer = SimpleCalcuLater(Question.Operation, Question.Number1, Question.Number2);
		Question.QuestionLevel = QuestionLevel;
		break;
	}

	return Question;
}

void GenerateQuizzQuestions(stQuizz& Quizz)
{
	for (short QuestionNumber = 0;QuestionNumber < Quizz.NumberOfQuestions;QuestionNumber++)
	{
		Quizz.QuestionList[QuestionNumber] = GenerateQuestion(Quizz.Operation, Quizz.QuestionLevel);
	}
}

void PrintQuestion(stQuizz Quizz, short QuestionNumber)
{
	cout << "\nQuestion [" << QuestionNumber + 1 << "/" << Quizz.NumberOfQuestions << "] \n\n";
	cout << Quizz.QuestionList[QuestionNumber].Number1 << "\n";
	cout << Quizz.QuestionList[QuestionNumber].Number2 << " ";
	cout << GetOpTypeSymbol(Quizz.QuestionList[QuestionNumber].Operation);
	cout << "\n____________" << endl;
}

void CorrectTheAnswer(stQuizz& Quizz, short QuestionNumber)
{
	if (Quizz.QuestionList[QuestionNumber].USerGuess == Quizz.QuestionList[QuestionNumber].CorrectAnswer)
	{
		Quizz.QuestionList[QuestionNumber].QuestionResult = true;
		Quizz.NumberOfCorrect++;
		
		MyLib::SetTextColor(2);
		cout << "Correct Answer :-)" << endl;
		MyLib::ResetTextColor();
	}
	else {
		Quizz.QuestionList[QuestionNumber].QuestionResult = false;
		Quizz.NumberOfWrong++;

		MyLib::SetTextColor(4);
		cout << "Wrong Answer!" << endl;
		MyLib::SetTextColor(6);
		cout << "Correct answer is: " << Quizz.QuestionList[QuestionNumber].CorrectAnswer << endl;
		MyLib::ResetTextColor();
	}
}

void AskAndCorrectAnswer(stQuizz &Quizz)
{
	for (short QuestionNumber = 0;QuestionNumber < Quizz.NumberOfQuestions;QuestionNumber++)
	{
		PrintQuestion(Quizz, QuestionNumber);
		Quizz.QuestionList[QuestionNumber].USerGuess = MyLib::ReadNumber();
		CorrectTheAnswer(Quizz, QuestionNumber);

	}

	Quizz.FinalResult = (Quizz.NumberOfCorrect >= Quizz.NumberOfWrong);
}

string GetFinalResultText(bool Result)
{
	if (Result)
	{
		MyLib::SetTextColor(2);
		return "Passed";
	}
	else
	{
		MyLib::SetTextColor(4);
		return "Failed";
	}
		
}

void PrintQuizzResult(stQuizz Quizz)
{
	cout << "\n\n---------------------\n";
	cout << "Result : " << GetFinalResultText(Quizz.FinalResult) << endl;
	MyLib::ResetTextColor();
	cout << "_____________________\n";
	cout << "Total Questions  : " << Quizz.NumberOfQuestions << "\n";
	cout << "Quiz level       : " << GetQuestionLevelText(Quizz.QuestionLevel) << "\n";
	cout << "OpType           : " << GetOpTypeName(Quizz.Operation) << "\n";
	cout << "Correct answers  : " << Quizz.NumberOfCorrect << "\n";
	cout << "Wrong answers    : " << Quizz.NumberOfWrong << "\n";
	cout << "_____________________\n";
}

string Tabs(short HowManyTabs)
{
	string t = "";

	for (int i = 0;i < HowManyTabs;i++)
	{
		t += "\t";
	}
	return t;
}

}

void PrintWrongSumaryy(stQuizz Quizz)
{
	if (Quizz.NumberOfWrong == 0)
	{
		cout << endl;
		cout << Tabs(4) << "You have (0) Wrong answers\n";
		cout << Tabs(4) << "Great Job!" << endl;
	}
	

	for (short QuestionNumber = 0;QuestionNumber < Quizz.NumberOfQuestions;QuestionNumber++)
	{
		if (Quizz.QuestionList[QuestionNumber].QuestionResult == false)
		{
			cout << endl;
			cout << Tabs(4) << "Question Number: " << QuestionNumber + 1 << endl;

			cout << Tabs(4) << "Your Answer    : ";
			MyLib::SetTextColor(4);
			cout << Quizz.QuestionList[QuestionNumber].USerGuess << endl;
			MyLib::ResetTextColor();
			cout << Tabs(4) << "Correct answer : ";
			MyLib::SetTextColor(2);
			cout << Quizz.QuestionList[QuestionNumber].CorrectAnswer << endl;
			MyLib::ResetTextColor();
			
		}
		
	}
}
